Here's a typed helper with capped exponential backoff and full jitter:
typescript
async function retry<T>(
  fn: () => Promise<T>, max = 5
): Promise<T> {
  for (let i = 0; ; i++) {
    try { return await fn(); }
    catch (e) {
      if (i >= max) throw e;
      const d = Math.min(1000 * 2 ** i, 15000);
      await sleep(Math.random() * d);
    }
  }
}
